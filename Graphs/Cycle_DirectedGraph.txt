#include<iostream>
#include<list>

#define NIL -1

using namespace std;

// Detecting cycle in directed Graph using DFS and recursion stack
class Graph
{
    int v;
    list<int> *adj;
    bool isCyclicUtil(int u, bool visited[], bool recStack[]);
    
    public:
    Graph(int v);
    void AddEdge(int u, int v);
    bool IsCyclic();
};

Graph :: Graph(int v)
{
    this->v = v;
    adj = new list<int>[v];
}

void Graph :: AddEdge(int u, int v)
{
    adj[u].push_back(v);
}

bool Graph :: isCyclicUtil(int u, bool visited[], bool recStack[])
{
    visited[u] = true;
    recStack[u] = true;
    
    list<int> :: iterator i;
    for(i = adj[u].begin(); i != adj[u].end(); ++i)
    {
        if(visited[*i] == false  && isCyclicUtil(*i, visited, recStack))
            return true;
        else if(visited[*i] && recStack[*i])
            return true;
    }
    recStack[u] = false; // Remove the vertex from the recusrion stack for parent calls in DFS tree
    
    return false;
}

bool Graph :: IsCyclic()
{
    bool *visited = new bool[v];
    bool *recStack = new bool[v];

    for(int i = 0; i < v; ++i)
    {
        visited[i] = false;
        recStack[i] = false;
    }
    
    for(int i = 0; i < v; ++i)
        if(visited[i] == false && isCyclicUtil(i, visited, recStack))
            return true;
    
    return false;
}

// Driver program to test above functions 
int main() 
{ 
    // Create a graph given in the above diagram 
    Graph G(6); 
    G.AddEdge(0, 1); 
    G.AddEdge(1, 2); 
    G.AddEdge(2, 0); 
    G.AddEdge(3, 4); 
    G.AddEdge(4, 5); 
  
    if (G.IsCyclic()) 
        cout << "Yes"; 
    else
        cout << "No"; 
  
    return 0; 
}
